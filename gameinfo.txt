"GameInfo"
{
	game 		"Portal 2: Community Edition - Archipelago"

	// The icon used in Steam.
	// Even though `icon` is set to "resource/icon", Steam will still use "resource/icon_big.tga"
	GameLogo	1
	icon		"resource/icon"
	
	// FGD file used by Hammer vbsp to automatically apply correct KVs
	GameData	"p2ce.fgd"
	
	// Set this to 1 to enable Workshop addon integration
	// This is disabled by default to prevent WS addons from being mounted in mods
	WorkshopEnabled 0

	// Mounts content from Portal 2 and P2CE.
	// To add additional mounts, see cfg/mounts.kv

	mount
	{
		// Portal 2
		620
		{
			"required" "1"
			
			// Base Game
			"portal2"
			{
				"vpk" "pak01"
				"dir" "."
			}
		}
	}

	FileSystem
	{
		SteamAppId				440000		// Required to identify as a P2CE mod. Replace this with your own AppID if you are publishing a mod on Steam.
		
		// If the user has a non-English language selected, for each "Game" path and mounted folder, another path with _<language>
		// at the end is automatically added. For example, 'p2ce' on a French-localized machine would have
		// 'p2ce_french' automatically mounted on top of the existing mod.

		// Search paths are relative to the base directory, which is 2 folders up from the P2:CE launcher executable (bin/{win64|linux64}/p2ce).
		// |gameinfo_path| points at the directory where gameinfo.txt is.
		// This is used in mods so the user can install the mod to a different location/drive and still have it work.

		SearchPaths
		{
			// This is the path used to load required P2:CE game libraries (client/server and coop matchmaking code)
			gamebin									p2ce/bin
			
			// Custom content and VPK assets
			game 									"|gameinfo_path|custom/*"
			
			// Mod core content will be mounted using this parameter, and all configuration files get written to this directory.
			game+mod+default_write_path				|gameinfo_path|.
			
			// Dependency content required to boot into this mod. "p2ce" must be added as game+mod,
			// because steam.inf and some other important files are pulled from the MOD path.
			// MOD may have multiple directories on it, hence why this works.
			game+mod								p2ce
			game									portal2
		}
	}
}
